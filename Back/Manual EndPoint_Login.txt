codigo para usar fetch puro:

let headersList = {
 "Accept": "*/*",
 "User-Agent": "Thunder Client (https://www.thunderclient.com)",
 "Content-Type": "application/json"
}

let bodyContent = JSON.stringify({
  "username":"principal",
  "password":"1234"
});

let response = await fetch("localhost:5002/api/login/", { 
  method: "POST",
  body: bodyContent,
  headers: headersList
});

let data = await response.text();
console.log(data);

codigo para usar axios:

import axios from "axios";

let headersList = {
 "Accept": "*/*",
 "User-Agent": "Thunder Client (https://www.thunderclient.com)",
 "Content-Type": "application/json" 
}

let bodyContent = JSON.stringify({
  "username":"principal",
  "password":"1234"
});

let reqOptions = {
  url: "localhost:5002/api/login/",
  method: "POST",
  headers: headersList,
  data: bodyContent,
}

let response = await axios.request(reqOptions);
console.log(response.data);

codigo para usar XMLHttpRequest:
const xhr = new XMLHttpRequest();
xhr.withCredentials = true;

xhr.addEventListener("readystatechange", function () {
  if (this.readyState === this.DONE) {
    console.log(this.responseText);
  }
});

xhr.open("POST", "localhost:0.0.19.138/api/login/");
xhr.setRequestHeader("Accept", "*/*");
xhr.setRequestHeader("User-Agent", "Thunder Client (https://www.thunderclient.com)");
xhr.setRequestHeader("Content-Type", "application/json");

xhr.send(data);
